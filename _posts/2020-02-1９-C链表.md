# 单向链表
1. 插入时O(1)，查找某一个节点O(n)

```c
#include<stdio.h>
#include<malloc.h>
#include<string.h>
#include<Windows.h>

typedef struct _node
{
	int value;
	struct _node* next;
}Node, * pNode;

pNode creat_simplest_list(int nLength)
{
	pNode node = (pNode)malloc(sizeof(Node));
	pNode temp = node;
	if (node == NULL)
	{
		return NULL;
	}
	memset(node, 0, sizeof(Node));
	for (int i = 0; i < nLength; i++)
	{
		if (i == nLength)
		{
			pNode node = NULL;
		}
		else
		{
			node->value = i * i;
			node->next = (pNode)malloc(sizeof(Node));
			if (node->next == NULL)
			{
				return NULL;
			}
			memset(node->next, 0, sizeof(Node));
			node = node->next;
		}
	}
	return temp;
}

void Test(void)
{
	pNode node = creat_simplest_list(10);
	for (int j = 0; j < 10; j++)
	{
		if (node != NULL)
		{
			printf("%d\n", node->value);
			node = node->next;
		}
	}
	free(node);
}

int main(void)
{
	Test();
}
```
