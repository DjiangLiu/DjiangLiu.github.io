# 裸函数
在函数名前面加上  ```__deplspec(naked)```，此时，编译器对该函数不会进行任何处理,即使是裸函数也要有```ret```,否则程序报错。如果只定义只是空函数，依然会生成一段汇编数据。
```
void func(){}
```	
![C空函数](/images/cc/2.png)
```
#include <stdio.h>

void __declspec(naked) Plus(int x,int y)
{
	__asm
	{
		//参数
		//局部变量，ebp-4
		//返回值，eax或者内存中

		//保留调用前的栈地
		push ebp
		//提升堆栈
		mov ebp,esp
		sub esp,0x40
		//保留现场
		push ebx
		push esi
		push edi
		//开始填充缓冲区
		mov eax,0xcccccccc
		mov ecx,0x10
		lea edi,dword ptr ds:[ebp-0x40]
		rep stosd

		//函数的和兴功能
		mov eax,dword ptr ds:[ebp+0x8]
		add eax,dword ptr ds:[ebp+0xc]

		//恢复现场
		pop edi
		pop esi
		pop ebx

		//降低堆栈
		mov esp,ebp
		pop ebp

		ret
	}
}

void __declspec(naked) Plus2(int x, int y,int z)
{
	__asm
	{
		//参数
		//局部变量，ebp-4
		//返回值，eax或者内存中

		//保留调用前的栈地
		push ebp
		//提升堆栈
		mov ebp, esp
		sub esp, 0x40
		//保留现场
		push ebx
		push esi
		push edi
		//开始填充缓冲区
		mov eax, 0xcccccccc
		mov ecx, 0x10
		lea edi, dword ptr ds : [ebp - 0x40]
		rep stosd

		//函数的主要功能
		//存储临时变量
		mov dword ptr ds : [ebp - 0x4], 0xa
		mov dword ptr ds : [ebp - 0x8], 0xb
		mov dword ptr ds : [ebp - 0xc], 0xc
		
		mov eax, dword ptr ds : [ebp + 0x8]
		add eax, dword ptr ds : [ebp + 0xc]
		add eax, dword ptr ds : [ebp + 0x10]
		add eax, dword ptr ds : [ebp - 0x4]
		add eax, dword ptr ds : [ebp - 0x8]
		add eax, dword ptr ds : [ebp - 0xc]



		//恢复现场
		pop edi
		pop esi
		pop ebx

		//降低堆栈
		mov esp, ebp
		pop ebp

		ret
	}
}

void Plus3(){}

void main() {
	Plus(1, 2);
	Plus2(1, 2, 3);
	Plus3();
}

```
