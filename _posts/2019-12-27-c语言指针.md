# 第一节
## 1.指针加减运算
```c
//主要是注意区域是否可写
#include <stdio.h>

int main(void) {
	const char *p = "test1234";
	printf("%c\n", *p++);
	//*(p++); p+=1

	//printf("%c\n", (*p)++);//err
	//*p;(*p)+=1
	
    printf("%c\n", *++p);
	//*（++p）; p+=1 ; *p
	
    //printf("%c\n", ++*p);
	//与++（*p）等价*p += 1；*p
	return 0;
}
```
##  数组名就是一个常量指针
```
int a1[10];
int a2[10][20];

a1+1;
&a1+1;//这个+1 是单位长度是去掉*后的长度

a2+1;
&a2+1;

a1 int * const a:4
&a1 int (* const a1)[10]:40

a2 int (* const a2)[20]: 20*4 : 80
&a2:int (* const a2)[10][20]: 10*20*4 800
```
