# 第一节
## 1.指针加减运算
```c
//主要是注意区域是否可写
#include <stdio.h>

int main(void) {
	const char *p = "test1234";
	printf("%c\n", *p++);
	//*(p++); p+=1

	//printf("%c\n", (*p)++);//err
	//*p;(*p)+=1
	
    printf("%c\n", *++p);
	//*（++p）; p+=1 ; *p
	
    //printf("%c\n", ++*p);
	//与++（*p）等价*p += 1；*p
	return 0;
}
```

# 作业
-   1.p是一个整型指针，比较下面语句的含义
```c
#include <stdio.h>

int main(void) {
	int a = 1234;
	int* p = &a;
	printf("%d\n", *(p + 1));
	//指针向后移动四个字节，并取四个字节
	printf("%d\n", *(char*)(p + 1));
	//指针向后移动四个字节，但取一个字节
	printf("%d\n", *(unsigned long*)(p + 1));
	//long 四字节， 指针向后移动四字节，取四字节，转为无符号数
	return 0;
}
```
