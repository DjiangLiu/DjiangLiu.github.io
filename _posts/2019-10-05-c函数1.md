#include <stdio.h>

int max(int x,int y)
{
    //int是返回值类型，max是函数名称
    //x,y是参数，已经类型
    //return 在汇编中就是放在eax，或者rax中
    return x>y?x:y;
}
//函数的声明（可以多次声明）
int add(int,int);

bool is_leap_year(int year)
{
    return ((year%4==0)&&(year%100!=0||year%400==0));
}

char mytolower(char ch)
{
    if (ch>='A'&&ch<='Z')
    {
        ch += 'a' - 'A';
    }
    return ch;
}


//需要注意在函数传递参数的时候，修改的内容是否发生变化。后续整理
void myswap1(int x,int y)
{
    int tmp = x;
    x = y;
    y = tmp;
}

void myswap2(int &x,int &y)
{
    x = x^y;
    y = x^y;
    x = x^y;
}

int main(void)
{
    //一个函数只实现一个功能
    printf("%d\n",max(100,11));
    int result = 0;
    int a = 100;
    int b = 200;
    myswap1(a,b);
    //result = add(100,50);
    result = add(a,b);
    printf("%d\n",add(1,2));
    printf("%d\n",result);
    getchar();
    return 0;
}

//函数的定义，只能定义一次
int add(int x,int y)
{
    int res = x + y ;
    return res;
}

#include <stdio.h>

int my_strlen(char *str)
{
    int len = 0;
    char *s = str;
    while (*s!='\0')
    {
        *s++;
        len++;
    }
    return len;
}

void reverse_str(char * str)
{
    int len = my_strlen(str);    
    for (int i = 0; i < len/2; i++)
    {
        str[i] = str[i] ^ str[len-1-i];
        str[len-1-i] = str[i] ^ str[len-1-i];
        str[i] = str[i] ^ str[len-1-i];
    }
    printf("%s\n",str);
    

}

int main(void)
{

    char str[] = "hello world 0987654321";
    reverse_str(str);
    getchar();
    return 0;
}
