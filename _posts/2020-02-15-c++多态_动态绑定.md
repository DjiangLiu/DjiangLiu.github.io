# 动态绑定
1. 只有virtual函数是动态绑定的
2. 动态绑定又叫多态

```c
#include<stdio.h>

class Base
{
public:
	int x;
public:
	Base()
	{
		x = 100;
	}
	void func1()
	{
		printf("Base func1\n");
	}
	virtual void func2()
	{
		printf("Base func2\n");
	}
};

class Sub :public Base
{
public:
	int x;
public:
	Sub()
	{
		x = 200;
	}
	void func1()
	{
		printf("Sub func1\n");
	}
	virtual void func2()
	{
		printf("Sub func2\n");
	}
};

void Test(Base* pb)
{
	int x = pb->x;
	printf("%d\n", x); // 前期绑定，编译期绑定,值、普通的函数
	pb->func1();//运行期间绑定 动态绑定 晚绑定
	pb->func2();//同一个函数不同的行为称为多态
}

int main(void)
{
	Sub b;
	Test(&b);
}

	Test(&b);
```

# 练习

```c
#include<stdio.h>

class Base
{
public:
	int x;
	int y;
public:
	Base()
	{
		x = 1;
		y = 2;
	}
	void Printf()
	{
		printf("Base %d %d\n", x, y);
	}
};

class Sub1 :public Base
{
public:
	int A;
public:
	Sub1()
	{
		x = 4;
		y = 5;
		A = 6;
	}
	void Printf()
	{
		printf("Sub1 %d %d  %d\n", x, y, A);
	}
};

class Sub2 :public Base
{
public:
	int B;
public:
	Sub2()
	{
		x = 7;
		y = 8;
		B = 9;
	}
	void Printf()
	{
		printf("Sub2 %d %d  %d\n", x, y, B);
	}
};

void Test()
{
	Base b1;
	Sub1 s1;
	Sub2 s2;

	Base* arr[] = { &b1,&s1,&s2 };
	for (int i = 0; i < 3; i++)
	{
		arr[i]->Printf();
	}
}

int main(void)
{
	Test();
}
```
