# 虚函数：间接调用

## 测试：
1. 直接调用间接调用

    call 地址   直接调用    E8

2. virtual虚函数 关键词

4. 通过指针调用，生成的是间接call

1. 添加virtual 函数结构体大小多出来四个字节.
2. 多出来的四个字节，就是结构体的前四个字节（首地址）（本身是一个地址）。
3. 多出来的四个字节，指向一个数组。这个数组就是虚函数表。里面存储了所有虚函数的地址。
4. 重载：名称一样，参数不一样。同作用域下，同函数名，参数不同（包括类型、个数、顺序）
5. 重写：名称、参数一样，子类重写父类

```c
#include<stdio.h>
#include<malloc.h>
#include<string.h>
#include<Windows.h>

class Base
{
public:
	int x;
	int y;
	Base()
	{
		x = 1; 
		y = 2;
	}
	virtual void func1()
	{
		printf("func1\n");
	}
	virtual void func2()
	{
		printf("func2\n");
	}
};

void Test(void)
{
	Base base;
	printf("base 的this地址为：%x\n", &base);
	printf("base 的虚函数表地址为：%x\n", *(int*)&base);
	printf("base 的x为：%x\n", *((int*)(&base) + 1));
	printf("base 的y为：%x\n", *((int*)(&base) + 2));

	typedef void(*pFunction)(void);
	pFunction pFun;
	pFun = (pFunction)(((int*)(*(int*)&base))[0]);
	pFun();
	pFun = (pFunction)(*((int*)(*(int*)&base) + 1));
	pFun();

}

int main(void)
{
	Test();
}
```

## 总结：
1. 通过对象调用时，virtual函数和普通函数都是E8 call（直接调用）
2. 通过指针调用时，virtual函数使用FF call（间接调用）

1. 单继承无函数覆盖。因为Sub:Base所以 虚函数表为从Base继承的函数开始



```c
#include<stdio.h>
#include<malloc.h>
#include<string.h>
#include<Windows.h>

struct Base
{
public:
	virtual void func1()
	{
		printf("func1\n");
	}
	virtual void func2()
	{
		printf("func2\n");
	}
	virtual void func3()
	{
		printf("func3\n");
	}
};

struct Sub:Base
{
public:
	virtual void func4()
	{
		printf("func4\n");
	}
	virtual void func5()
	{
		printf("func5\n");
	}
	virtual void func6()
	{
		printf("func6\n");
	}
};


void Test(void)
{
	Sub s1;
	printf("s1的首地址：%x\n", &s1);
	printf("s1的虚函数表：%x\n", *(int*)&s1);
	typedef void(*pFunc)(void);
	pFunc pf;
	pf = (pFunc)*((int*)(*(int*)&s1) + 1);
	pf();
}

int main(void)
{
	Test();
}
```

2. 单继承有函数覆盖。因为func1和func2被重写，所以，虚函数表示覆盖原来func1和func2的位置。

```c
#include<stdio.h>
#include<malloc.h>
#include<string.h>
#include<Windows.h>

struct Base
{
public:
	virtual void func1()
	{
		printf("func1\n");
	}
	virtual void func2()
	{
		printf("func2\n");
	}
	virtual void func3()
	{
		printf("func3\n");
	}
};

struct Sub:Base
{
public:
	virtual void func1()
	{
		printf("sub : func4\n");
	}
	virtual void func2()
	{
		printf("sub : func5\n");
	}
	virtual void func6()
	{
		printf("func6\n");
	}
};


void Test(void)
{
	Sub s1;
	printf("s1的首地址：%x\n", &s1);
	printf("s1的虚函数表：%x\n", *(int*)&s1);
	typedef void(*pFunc)(void);
	pFunc pf;
	pf = (pFunc)*((int*)(*(int*)&s1) + 3);
	pf();
}

int main(void)
{
	Test();
}
```
